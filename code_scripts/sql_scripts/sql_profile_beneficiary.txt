-- [18] The Composite Data Quality Risk Score
-- This creates a single "Risk Score" by aggregating multiple anomaly percentages. It helps immediately flag the dirtiest fields across the board without having to look at a dozen different queries.
SELECT 
    file_name,
    field_name,
    rec_cnt,
    (CAST(null_cnt_pct AS FLOAT) + 
     CAST(space_cnt_pct AS FLOAT) + 
     CAST(ledspc_pct AS FLOAT) + 
     CAST(spchr_cnt_pct AS FLOAT)) AS total_risk_score
FROM raw_cms_profile
WHERE cat1 = 'beneficiary'
ORDER BY total_risk_score DESC;


-- [19] Data Type Collision (The "Split Personality" Check)
-- This isolates fields that are highly populated but split heavily between numeric and alphanumeric formats. This usually indicates unstandardized data entry, mixed encoding, or shifting upstream schemas.
SELECT 
    file_name,
    field_name,
    rec_cnt,
    alpnmc_cnt_pct,
    numrc_cnt_pct
FROM raw_cms_profile
WHERE cat1 = 'beneficiary'
  AND CAST(alpnmc_cnt_pct AS FLOAT) > 10.0 
  AND CAST(numrc_cnt_pct AS FLOAT) > 10.0
  AND (CAST(alpnmc_cnt_pct AS FLOAT) + CAST(numrc_cnt_pct AS FLOAT)) > 80.0
ORDER BY rec_cnt DESC;


-- [20] The "Suspiciously Perfect" Identifier Check
-- This flags fields that claim to be nearly 100% unique but have wild length variances. It catches standard IDs that have been padded, corrupted, or mixed with completely different identifier formats.
SELECT 
    file_name,
    field_name,
    rec_cnt,
    uniqu_pct,
    min_len_pct,
    max_len_pct
FROM raw_cms_profile
WHERE cat1 = 'beneficiary'
  AND CAST(uniqu_pct AS FLOAT) > 95.0
  AND (CAST(min_len_pct AS FLOAT) < 50.0 AND CAST(max_len_pct AS FLOAT) < 50.0)
ORDER BY CAST(uniqu_pct AS FLOAT) DESC;


-- [21] High Volume, Zero Nulls, Low Cardinality (The "Default Value" Trap)
-- This finds fields that are fully populated but contain very few distinct values relative to a massive record count. This is a classic symptom of a system forcing a hardcoded default (like 'N/A' or '0') into a field that should legitimately have nulls.
SELECT 
    file_name,
    field_name,
    rec_cnt,
    null_cnt_pct,
    uniqu
FROM raw_cms_profile
WHERE cat1 = 'beneficiary'
  AND CAST(null_cnt_pct AS FLOAT) = 0.0
  AND CAST(uniqu AS INT) < 5
  AND CAST(rec_cnt AS INT) > 10000
ORDER BY CAST(uniqu AS INT) ASC;
